import random # Sample text corpus text = """Machine learning is amazing. Text generation using Markov chains is fun. Markov chains create new sentences based on probability.""" # Step 1: Tokenize the text into words words = text.split() # Step 2: Build the Markov chain model markov_chain = {} for i in range(len(words)-1): current_word = words[i] next_word = words[i+1] if current_word not in markov_chain: markov_chain[current_word] = [] markov_chain[current_word].append(next_word) # Step 3: Generate text using the model def generate_text(chain, start_word, length=15): word = start_word sentence = [word] for _ in range(length-1): if word not in chain: break next_word = random.choice(chain[word]) sentence.append(next_word) word = next_word return " ".join(sentence) # Example Run print("Generated Text:\n") print(generate_text(markov_chain, "Markov", 20))
